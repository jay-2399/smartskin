Create `src/features/funnel/validation.ts` :
```ts
import type { Answers, StepId } from "./types";
import { QUESTIONS } from "./questions";

/** Bascule une valeur dans une sélection multi, en gérant l'exclusivité.
 *  - cocher une option `exclusive` → la sélection ne contient plus qu'elle
 *  - cocher une option normale → retire toute option exclusive précédemment posée */
export function toggleOption(
  current: string[],
  value: string,
  isExclusive: boolean
): string[] {
  if (current.includes(value)) return current.filter((v) => v !== value);
  if (isExclusive) return [value];
  // retire les exclusives connues (celles marquées exclusive dans une question)
  const exclusiveValues = new Set(
    Object.values(QUESTIONS)
      .flatMap((q) => [...q.options, ...(q.revealOptions ?? [])])
      .filter((o) => o.exclusive)
      .map((o) => o.value)
  );
  const cleaned = current.filter((v) => !exclusiveValues.has(v));
  return [...cleaned, value];
}

export function isStepValid(step: StepId, answers: Answers): boolean {
  switch (step) {
    case "q1": return answers.q1.length > 0;
    case "q2": return answers.q2.length > 0;
    case "q3": return answers.q3.length > 0;
    case "q4": return answers.q4 !== null;
    case "q5":
      if (answers.q5.changed === false) return true;
      if (answers.q5.changed === true) return answers.q5.symptoms.length > 0;
      return false;
    case "q6": return answers.q6 !== null;
    case "q7": return answers.q7.length > 0;
  }
}
```

```ts
import { create } from "zustand";
import type { Answers, StepId } from "./types";
import { EMPTY_ANSWERS } from "./types";
import { toggleOption } from "./validation";

type FunnelState = {
  answers: Answers;
  photo: Blob | null;
  setSingle: (step: "q4" | "q6", value: string) => void;
  toggleMulti: (step: "q1" | "q2" | "q3" | "q7", value: string, exclusive: boolean) => void;
  setGate: (changed: boolean) => void;
  toggleSymptom: (value: string) => void;
  setPhoto: (blob: Blob | null) => void;
  reset: () => void;
};

export const useFunnel = create<FunnelState>((set) => ({
  answers: structuredClone(EMPTY_ANSWERS),
  photo: null,
  setSingle: (step, value) =>
    set((s) => ({ answers: { ...s.answers, [step]: value } })),
  toggleMulti: (step, value, exclusive) =>
    set((s) => ({
      answers: { ...s.answers, [step]: toggleOption(s.answers[step], value, exclusive) },
    })),
  setGate: (changed) =>
    set((s) => ({
      answers: { ...s.answers, q5: { changed, symptoms: changed ? s.answers.q5.symptoms : [] } },
    })),
  toggleSymptom: (value) =>
    set((s) => ({
      answers: {
        ...s.answers,
        q5: { ...s.answers.q5, symptoms: toggleOption(s.answers.q5.symptoms, value, false) },
      },
    })),
  setPhoto: (blob) => set({ photo: blob }),
  reset: () => set({ answers: structuredClone(EMPTY_ANSWERS), photo: null }),
}));
```

```tsx
"use client";
import { useRouter } from "next/navigation";
import { QUESTIONS, STEP_ORDER } from "@/features/funnel/questions";
import { useFunnel } from "@/features/funnel/store";
import { isStepValid } from "@/features/funnel/validation";
import { OptionList } from "@/components/ui/OptionList";
import { ProgressBar } from "@/components/ui/ProgressBar";
import type { StepId } from "@/features/funnel/types";

export function QuestionScreen({ step }: { step: StepId }) {
  const router = useRouter();
  const q = QUESTIONS[step];
  const answers = useFunnel((s) => s.answers);
  const { setSingle, toggleMulti, setGate, toggleSymptom } = useFunnel.getState();

  const valid = isStepValid(step, answers);

  const next = () => {
    const i = STEP_ORDER.indexOf(step);
    if (i < STEP_ORDER.length - 1) router.push(`/questions/${STEP_ORDER[i + 1]}`);
    else router.push("/compte"); // mur d'inscription (Plan 4)
  };

  return (
    <div className="qscreen">
      <ProgressBar index={q.index} />
      <h2 className="qtitle">{q.title}</h2>

      {q.mode === "single" && (
        <OptionList
          options={q.options}
          selected={answers[step as "q4" | "q6"] ? [answers[step as "q4" | "q6"] as string] : []}
          onToggle={(v) => setSingle(step as "q4" | "q6", v)}
        />
      )}

      {q.mode === "multi" && (
        <OptionList
          options={q.options}
          selected={answers[step as "q1" | "q2" | "q3" | "q7"]}
          onToggle={(v) =>
            toggleMulti(
              step as "q1" | "q2" | "q3" | "q7",
              v,
              !!q.options.find((o) => o.value === v)?.exclusive
            )
          }
        />
      )}

      {q.mode === "gate" && (
        <>
          <OptionList
            options={q.options}
            selected={answers.q5.changed === true ? ["yes"] : answers.q5.changed === false ? ["no"] : []}
            onToggle={(v) => setGate(v === "yes")}
          />
          {answers.q5.changed === true && q.revealOptions && (
            <OptionList options={q.revealOptions} selected={answers.q5.symptoms} onToggle={toggleSymptom} />
          )}
        </>
      )}

      <button type="button" className="cta" disabled={!valid} onClick={next}>
        Continuer
      </button>
    </div>
  );
}
```
